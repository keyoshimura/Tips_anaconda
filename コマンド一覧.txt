
●anaconda本体のupdate
conda update conda

●パッケージ一式のupdate
conda update --all

●pythonのヴァージョンのupdate
# 3.5にupdate
conda update python=3.5

●仮想環境の構築
# hogeという名前の環境をpython3.5系で作成
# 最後に「anaconda」をつけるとanacondaをが管理しているパッケージ全てが入る
conda create --name hoge python=3.5 anaconda


●環境の一覧を取得
conda info -e

●仮想環境に入る
# hogeという環境に入る
activate hoge
source activate hoge

●仮想環境から出る
deactivate

●仮想環境の削除
# hogeという環境を削除する
conda remove --name hoge --all


●パッケージ一覧の確認
# 対象の環境に入ってから下記のコマンド
conda list

●パッケージのインストール
# sklearnのインストール
conda install sklearn
